            return counter += 1;
        };

    }

    alert(counter());
    alert(counter());
    alert(counter());
    alert(counter());"
[Output:1,2,3,4]

        (function () {

            let count = 1;
        console.log(count); // What is logged?
            }
        console.log(count); // What is logged?
    })();"
[Output:1,0]
Explanation: when console.log(count) was count first time, there it matched the if condition, and when it was called second time it searched the value which was out of the function scope so it accessed the let variable of count i.e 0.
